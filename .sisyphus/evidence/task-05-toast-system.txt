## Usage Example

```tsx

function MyComponent() {
  const { toast } = useToast()
  
  return (
    <button onClick={() => toast({
      title: 'Success',
      description: 'Transaction submitted',
      variant: 'success'
    })}>
      Show Toast
    </button>
  )
}
```
